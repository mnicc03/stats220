My meme

  • The meme I created shows the honest reaction of all gym goers when their training partner bails out of a gym session, especially after taking pre-workout which is only meant to be used for serious sessions, using Patrick Bateman (played by Christian Bale).

library(magick)

#Grabbing Christian Bale images and scaling to 400
happy_bale <- image_read("https://i.ytimg.com/vi/NStwCKpS-Eg/hqdefault.jpg") %>%
  image_scale(400)

disappointed_bale <- image_read("https://i.ytimg.com/vi/GzhOhQ_X6us/hqdefault.jpg") %>%
  image_scale(400)

#Formatting black squares with white text to be later paired with images 
happy_text <- image_blank(width = 400,
                          height = 300,
                          color = "#000000") %>%
  image_annotate(text = "Taking preworkout to get ready for the\nmost diabolical leg session\nknown to man with your gym buddy",
                 color = "#FFFFFF",
                 size = 20,
                 font = "Impact",
                 gravity = "center")

disappointed_text <- image_blank(width = 400,
                                 height = 300,
                                 color = "#000000") %>%
  image_annotate(text = "'Sorry bro, I actually\ncan't make it today'",
                 color = "#FFFFFF",
                 size = 40,
                 font = "Impact",
                 gravity = "center")
  
#Creating each row for the meme
top_row <- image_append(c(happy_bale, happy_text))

bottom_row <- image_append(c(disappointed_bale, disappointed_text))

meme <- c(top_row, bottom_row) %>%
  image_append(stack = TRUE)

meme

#Saving meme as png.

image_write(meme, "my_meme.png")

My animated GIF

  • The purpose of my GIF was to create a basic series of frames showing simple use of images and text.

library(magick)

#Gathering each circle from web
red_circle <- image_read("https://www.publicdomainpictures.net/pictures/310000/velka/red-circle.png") %>%
  image_scale(400)

blue_circle <- image_read("https://www.publicdomainpictures.net/pictures/310000/velka/blue-circle.png") %>%
  image_scale(400)

green_circle <- image_read("http://buzz.shiftingretail.com/wp-content/uploads/green_circle_png.png") %>%
  image_scale(400)

yellow_circle <- image_read("https://www.publicdomainpictures.net/pictures/310000/velka/yellow-circle.png") %>%
  image_scale(400)

#Creating each individual frame

frame1 <- red_circle %>%
  image_annotate(text = "RED",
                 size = 100,
                 gravity = "center") %>%
  image_scale(300) %>%
  image_extent("400x400")

frame2 <- blue_circle %>%
  image_annotate(text = "BLUE",
                 size = 100,
                 gravity = "center") %>%
  image_scale(300) %>%
  image_extent("400x400")

frame3 <- green_circle %>%
  image_annotate(text = "GREEN",
                 size = 90,
                 gravity = "center") %>%
  image_scale(300) %>%
  image_extent("400x400")

frame4 <- yellow_circle %>%
  image_annotate(text = "YELLOW",
                 size = 90,
                 gravity = "center") %>%
  image_scale(300) %>%
  image_extent("400x400")

#Using a vector to put the frames in order
frames <- c(frame1, frame2, frame3, frame4)

#Animating frames
animation <- image_animate(frames, fps = 1) 

animation

#Saving gif 
image_write(animation, "my_animation.gif")
